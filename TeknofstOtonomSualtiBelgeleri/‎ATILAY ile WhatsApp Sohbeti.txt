# BMX160 class definition
class BMX160:
    def __init__(self, bus, address):
        self.bus = bus
        self.address = address
        self.setup()

    def setup(self):
        self.bus.write_byte_data(self.address, 0x7E, 0x11)  # Accelerometer normal mode
        time.sleep(0.05)
        self.bus.write_byte_data(self.address, 0x7E, 0x15)  # Gyroscope normal mode
        time.sleep(0.05)
        self.bus.write_byte_data(self.address, 0x7E, 0x19)  # Magnetometer normal mode
        time.sleep(0.05)

    def read_data(self):
        data = self.bus.read_i2c_block_data(self.address, 0x04, 23)
        return data

# BMP388 class definition
class BMP388:
    def __init__(self, bus, address):
        self.bus = bus
        self.address = address
        self.setup()

    def setup(self):
        self.bus.write_byte_data(self.address, 0x1B, 0x30)  # Set normal mode, standby time = 0.5ms
        self.bus.write_byte_data(self.address, 0x1C, 0x33)  # Set oversampling settings
        self.bus.write_byte_data(self.address, 0x1D, 0x00)  # Set IIR filter to off

    def read_data(self):
        data = self.bus.read_i2c_block_data(self.address, 0x04, 6)
        return data
